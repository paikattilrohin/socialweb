DROP TABLE post;
DROP TABLE user;

CREATE TABLE user(userid INT NOT NULL AUTO_INCREMENT,
                  name VARCHAR(50) NOT NULL,
                  email VARCHAR(50) NOT NULL,
                  password VARCHAR(25) NOT NULL,
                  CONSTRAINT USER_PK PRIMARY KEY (USERID));

CREATE TABLE post(postid INT NOT NULL AUTO_INCREMENT,
                   title VARCHAR(255),
                   name varchar(20),
                   content VARCHAR(255) NOT NULL,
                   userid VARCHAR(255),
                   `date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
                   CONSTRAINT Post_PK PRIMARY KEY (postid) );

CREATE TABLE like(likeid INT NOT NULL AUTO_INCREMENT,
                  postid VARCHAR(20),
                  userid VARCHAR(20),
                  CONSTRAINT USER_PK PRIMARY KEY (likeid));

CREATE TABLE favorite(favoriteid INT NOT NULL AUTO_INCREMENT,
                  postid VARCHAR(20),
                  userid VARCHAR(20),
                  tag VARCHAR(50),
                  CONSTRAINT USER_PK PRIMARY KEY (favoriteid));
